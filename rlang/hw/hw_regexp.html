抽取字組

看看是不是#開頭的字串,如果是的話要置換,例如 "# hello" ->"# hello "

str<-"# hello"
if(grepl("^[#]* ",str)) {

x<-sub("(^[#]* )(.*$)","\\1 <font color='red'> \\2 </font>",str, perl=TRUE)
print(x)
}
#> [1] "#  <font color='red'> hello </font>"

類似上題

divtitle<-function (str){
  if(grepl("^[#]* ",str)) {
    rst<-sub("(^[#]* )(.*$)","\\1 <div class='outposix'> \\2 </div>",str, perl=TRUE)
  }else{
    rst<-paste0("<div class='outposix'>",str,"</div>")
  }
  
}

pattern<-"out"
ext<-"oo"
list<-list("out_中文_1.oo","out_1.oo","out_xx_1.oo","out_中文2_1.oo")
what<-paste0("(?<=",pattern,"_).*(\\d)+","(?=\\.",ext,")");

r1<-stringr::str_extract(list,what)
r1
#> [1] "中文_1"  "1"       "xx_1"    "中文2_1"
stringr::str_extract(r1,"\\d$")
#> [1] "1" "1" "1" "1"
