    u.test<-function(x,mu,thegma,alternative="twoside")
{  Se=thegma/sqrt(length(x))
   u <-(mean(x)-mu)/Se
   p <-pnorm(u)
   if(alternative=="twoside") p<-2*(1-pnorm(abs(u)))
   else if (alternative=="less") p<-pnorm(u)
   else p <- 1-pnorm(u)
   list(u=u, p=p)
}
