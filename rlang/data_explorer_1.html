data.frame 的record篩選，可以利用邏輯值索引來篩選分組，這裡的範例都是求得一組(以第一分類為例)：

ft<-factor(dframe$group)
gn<-length(levels(ft)) # nlevels(ft)

gf<-as.numeric(levels(ft)[1]) # 第一組
g1<-dframe[group %in% gf,] # 不是 gf %in% group
rst<-summary(g1)

ft<-factor(dframe$group)

  g1<-dframe[group %in% gf,] 
  t<-summary(g1)
  rst[gi]=t[4,1]

f1<-function(im)
{
