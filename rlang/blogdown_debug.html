    build_one<-function (input, output, to_md = file_ext(output) != "html", 
quiet = TRUE) 
{
options(htmltools.dir.version = FALSE)
setwd(dirname(input))
input = basename(input)
if (to_md) 
    options(bookdown.output.markdown = TRUE)
res = rmarkdown::render(input, "blogdown::html_page", output_file = output, 
    envir = globalenv(), quiet = quiet, run_pandoc = !to_md, 
    clean = !to_md) # 🏷
      👏res =>"D:/temp/xxxx/content/post/test.html"
x = read_utf8(res)
if (to_md) 
    x = process_markdown(res, x)
unlink(res)
x
}

    processmarkdown<-function (res, x = read_utf8(res)) 
{
unlink(xfun::attr(res, "intermediates"))
if (length(meta <- xfun::attr(res, "knit_meta"))) {
    m = rmarkdown:::html_dependencies_as_string(meta, attr(res, 
    "files_dir"), ".")
    if (length(i <- grep("^---\\s*$", x)) >= 2) {
    x = append(x, m, i[2])
    }
    else warning("Cannot find the YAML metadata in the .markdown output file. ", 
    "HTML dependencies will not be rendered.")
}
x = local({
    f = wd_tempfile(".md~", pattern = "post")
    on.exit(unlink(f), add = TRUE)
    write_utf8(x, f)
    bookdown:::process_markdown(f, "markdown", NULL, TRUE, 
    TRUE, x, NULL)
})
if (get_option("blogdown.protect.math", TRUE)) 
    x = xfun::protect_math(x)
x = gsub("<!--/?html_preserve-->", "", x)
if (run_pandoc(x)) {
    mds = replicate(2, wd_tempfile(".md~", pattern = "citation"))
    on.exit(unlink(mds), add = TRUE)
    write_utf8(x, mds[1])
    rmarkdown::pandoc_convert(mds[1], from = "markdown", 
    to = paste(markdown_format(), collapse = ""), output = mds[2], 
    options = c(if (!rmarkdown::pandoc_available("2.11.2")) "--atx-headers", 
        "--wrap=preserve"), citeproc = TRUE)
    x = c(bookdown:::fetch_yaml(x), "", read_utf8(mds[2]))
}
x
}
